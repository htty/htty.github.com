def escape_or_warn_of_escape_sequences(arguments)
  arguments.collect do |a|
    if a =~ /%[0-9a-f]{2}/i
      say "Argument '#{a}' was not URL-escaped because it contains escape " +
          'sequences'
      a
    else
      URI.escape a
    end
  end
end
