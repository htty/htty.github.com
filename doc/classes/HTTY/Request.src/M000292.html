  def address(address)
    uri = self.class.parse_uri(address)
    if response
      dup = dup_without_response
      return self.class.clear_cookies_if_host_changes(dup) do
        dup.uri = uri
      end
    end

    self.class.clear_cookies_if_host_changes self do
      @uri = uri
    end
  end
