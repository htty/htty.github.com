  def perform
    unless arguments.empty?
      raise ArgumentError,
            "wrong number of arguments (#{arguments.length} for 0)"
    end

    add_request_if_has_response do |request|
      notify_if_cookies_cleared request do
        request.follow session.last_response
      end
    end
  end
