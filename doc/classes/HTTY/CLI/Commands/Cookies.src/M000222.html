  def perform
    cookies = session.requests.last.cookies
    margin = cookies.inject 0 do |result, cookie|
      [cookie.first.length, result].max
    end
    cookies.each do |name, value|
      puts "#{name.rjust margin}: #{value}"
    end
    self
  end
