  def perform
    request = session.requests.last
    headers = if request.request_method.nil?
                if request.body.to_s.length.zero?
                  request.headers false
                else
                  request.headers true
                end
              else
                request.headers
              end
    show_headers headers, HTTY::Request::COOKIES_HEADER_NAME
    self
  end
