  def self.namespaces
    container = nil
    name.split('::')[0...-1].collect do |element|
      if container.nil?
        container = instance_eval("::#{element}", __FILE__, __LINE__)
      else
        container = container.module_eval(element, __FILE__, __LINE__)
      end
    end
  end
